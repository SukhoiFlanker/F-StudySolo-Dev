## 3. 具体重构方案

### 3.1 跨 Store 解耦

**问题：** `useAIChatStore.pushMessage` 直接调用 `useConversationStore.getState()`

**解决方案：在调用方处理同步，而不是 Store 内部**

```typescript
// features/workflow/hooks/use-stream-chat.ts

export function useStreamChat() {
  const pushMessage = useAIChatStore(s => s.pushMessage);
  const createConversation = useConversationStore(s => s.createConversation);
  const appendMessage = useConversationStore(s => s.appendMessage);
  const activeId = useConversationStore(s => s.activeId);

  const sendMessage = async (role: 'user' | 'assistant', content: string) => {
    // 在调用方显式处理跨 store 同步
    if (!activeId) createConversation();
    const id = pushMessage(role, content);
    appendMessage({ id, role, content, timestamp: Date.now() });
    return id;
  };

  return { sendMessage };
}
```

**或者：使用事件模式**

```typescript
// stores/chat/events.ts
export const CHAT_EVENTS = {
  MESSAGE_ADDED: 'chat:message-added',
} as const;

// useAIChatStore - 只更新自己的状态
pushMessage: (role, content) => {
  const entry = { id, role, content, timestamp: Date.now() };
  set(state => ({ history: [...state.history, entry] }));
  // 触发事件而不是直接调用另一个 store
  window.dispatchEvent(new CustomEvent(CHAT_EVENTS.MESSAGE_ADDED, { detail: entry }));
  return entry.id;
}

// useConversationStore - 监听事件
constructor() {
  window.addEventListener(CHAT_EVENTS.MESSAGE_ADDED, (e) => {
    this.appendMessage(e.detail);
  });
}
```

**推荐：方案一（调用方显式同步）**

#### admin/ 评估

**不建议拆分**，原因：
1. Admin 模块代码量适中
2. Admin 页面不会频繁新增
3. 拆分后维护成本增加

**建议优化：** 增加类型安全和注释，而不是拆分。

### 3.5 事件通信模式改进

**当前：** 散落的 CustomEvent

**改进：统一事件总线**

```typescript
// lib/events/event-bus.ts

type EventMap = {
  'canvas:tool-change': { tool: string };
  'canvas:node-placed': { nodeId: string };
  'workflow:execution-start': { workflowId: string };
  'panel:toggle': { panel: string };
};

class EventBus {
  on<K extends keyof EventMap>(event: K, handler: (data: EventMap[K]) => void) {
    window.addEventListener(event, (e) => handler((e as CustomEvent).detail));
  }

  emit<K extends keyof EventMap>(event: K, data: EventMap[K]) {
    window.dispatchEvent(new CustomEvent(event, { detail: data }));
  }
}

export const eventBus = new EventBus();
```

**使用：**
```typescript
// 发送
eventBus.emit('canvas:tool-change', { tool: 'select' });

// 接收
eventBus.on('canvas:tool-change', ({ tool }) => {
  // ...
});
```

**预计改动：** 2-3 小时，需要迁移所有 CustomEvent 调用。

### 5.3 插件注册机制

```typescript
// plugins/registry.ts

interface Plugin {
  id: string;
  name: string;
  nodeTypes: string[];
  renderer: React.ComponentType;
  configSchema?: Record<string, any>;
}

const plugins: Plugin[] = [];

// 动态加载插件
export function registerPlugin(plugin: Plugin) {
  plugins.push(plugin);
  // 注册到 RENDERER_REGISTRY
  RENDERER_REGISTRY[plugin.nodeTypes[0]] = plugin.renderer;
}

export function getPlugins(): Plugin[] {
  return [...plugins];
}
```

## 7. 风险评估

| Phase | 风险 | 缓解措施 |
|-------|------|---------|
| Store 重构 | 影响范围广 | 先在小功能试点 |
