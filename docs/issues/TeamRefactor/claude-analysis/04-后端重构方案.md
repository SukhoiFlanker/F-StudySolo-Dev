**步骤：**

1. 将 `ai_chat_stream.py` 中的共享函数移到新位置：

   ```python

   @router.post("/chat/stream")               # 流式
   ```

### 3.2 usage_ledger 模式统一

**装饰器方案：**

```python
# backend/app/services/usage/decorators.py

from functools import wraps
from .tracker import UsageTracker

def track_usage(source_type: str, source_subtype: str | None = None):
    """Decorator for tracking AI usage."""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            tracker = UsageTracker()

            # 从函数参数推断 user_id
            user = kwargs.get('current_user') or args[0] if args else None
            user_id = user.get('id') if user else None

            async with tracker.request(user_id, source_type, source_subtype) as req:
                try:
                    result = await func(*args, **kwargs)
                    req.set_status("completed")
                    return result
                except Exception as e:
                    req.set_status("failed")
                    raise
    return wrapper
```

**应用：**
```python
# before
@router.post("/chat")
async def ai_chat(body: AIChatRequest, current_user = Depends(...)):
    usage_request = await create_usage_request(...)
    with bind_usage_request(usage_request):
        ...

# after
@router.post("/chat")
@track_usage("assistant", "chat")
async def ai_chat(body: AIChatRequest, current_user = Depends(...)):
    # ... 业务逻辑 ...
```

**预计改动：**
- 新建 `services/usage/decorators.py`
- 修改 `api/ai.py`, `api/ai_chat.py`, `api/workflow_execute.py`
- 删除 `usage_ledger.py` 中的部分代码

```python
# backend/app/core/config.py

from pydantic import Field
from pydantic_settings import BaseSettings

class AIProviderConfig(BaseSettings):
    api_key: str = ""
    is_configured: bool = False

class Settings(BaseSettings):
    model_config = SettingsConfigDict(env_file=".env")

    # Database
    supabase_url: str
    supabase_anon_key: str
    supabase_service_role_key: str

    # AI Providers
    dashscope: AIProviderConfig = Field(default_factory=AIProviderConfig)

    siliconflow: AIProviderConfig = Field(default_factory=AIProviderConfig)

    # Email

        smtp_host: str = ""
        smtp_port: int = 587
        smtp_user: str = ""
        smtp_password: str = ""
```
